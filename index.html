Code: Quick Sensitivity (Python)


import numpy as np

# Earnings sensitivity to NIM (Net Interest Margin) shift
base_nim = 0.031   # 3.1%
assets   = 180e9   # interest-earning assets

shifts = np.array([-0.0005, 0.0, 0.0005, 0.0010])  # -5 to +10 bps

def earnings_delta(nim_shift):
    return assets * nim_shift

for s in shifts:
    delta = earnings_delta(s)
    print(f"NIM shift: {s*1e4:.0f} bps → ΔNI: ${delta/1e6:,.1f}M")

# Note: This ignores hedging/taxes for simplicity and is illustrative only.
